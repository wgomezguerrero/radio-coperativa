<script>
    // --- COLOCA AQUÍ TUS CREDENCIALES DE FIREBASE ---
    const firebaseConfig = {
  apiKey: "AIzaSyBT57itkr-WcslebPzggdinoPGqrJTUzGE",
  authDomain: "taxivellavega.firebaseapp.com",
  databaseURL: "https://taxivellavega-default-rtdb.europe-west1.firebasedatabase.app",
  projectId: "taxivellavega",
  storageBucket: "taxivellavega.firebasestorage.app",
  messagingSenderId: "627311271806",
  appId: "1:627311271806:web:30961d837fe32fa1ae9d2f"
};
    
    if (!firebase.apps.length) { firebase.initializeApp(firebaseConfig); }
    const db = firebase.database();

    let estaTransmitiendo = false;
    let miCodigoSocio = "Socio_" + Math.floor(Math.random() * 899 + 100);
    let ultimoAudioTimestamp = 0;
    
    let grabadorNativo = null;
    let fragmentosAudio = [];
    let streamMicrofono = null;

    const socioIdLogged = localStorage.getItem("socio_id_logged");
    if(socioIdLogged && socioIdLogged.trim() !== "") { miCodigoSocio = socioIdLogged; }

    const btnPtt = document.getElementById('btnPttRadio');
    const txtBotonPtt = document.getElementById('txtBotonPtt');
    const lblEstado = document.getElementById('lblEstadoConexion');
    const lblQuienHabla = document.getElementById('lblQuienHabla');

    window.onload = function() {
        // Al ser cargado desde una URL Web, esto disparará la pregunta nativa de Android de inmediato
        navigator.mediaDevices.getUserMedia({ audio: true })
            .then(stream => {
                streamMicrofono = stream;
                btnPtt.removeAttribute('disabled');
                txtBotonPtt.innerText = "PULSAR";
                lblEstado.innerText = "ONLINE - FRECUENCIA ABIERTA";
                lblQuienHabla.innerHTML = `<span style="color: var(--co-texto-secundario);"><i class="fa-solid fa-circle-check"></i> Frecuencia Lista (Silencio)</span>`;
            })
            .catch(e => {
                lblEstado.innerText = "SIN ACCESO AL MICRÓFONO";
                lblQuienHabla.innerHTML = `<span style="color:red;">Debes permitir el uso del micrófono en la barra superior.</span>`;
            });

        activarReceptorRadio();
    }

    function activarReceptorRadio() {
        db.ref('radio_frecuencia/canal_01').on('value', (snapshot) => {
            const data = snapshot.val();
            if (!data) return;
            if (data.usuario === miCodigoSocio) return;

            if (data.audioBase64 && data.timestamp && data.timestamp > ultimoAudioTimestamp) {
                ultimoAudioTimestamp = data.timestamp;
                if (!estaTransmitiendo) {
                    btnPtt.classList.add('recibiendo');
                    lblEstado.innerText = "RECIBIENDO AUDIO";
                    lblQuienHabla.innerHTML = `<span style="color:var(--co-exito); font-weight:600;"><i class="fa-solid fa-volume-high"></i> Escuchando...</span>`;
                    reproducirVozEntrante(data.audioBase64);
                }
            } 
            else if (data.estado && data.estado.transmitiendo) {
                if (!estaTransmitiendo && !btnPtt.classList.contains('recibiendo')) {
                    lblEstado.innerText = "CANAL OCUPADO";
                    lblQuienHabla.innerHTML = `<span style="color:var(--co-primario); font-weight:600;"><i class="fa-solid fa-headset"></i> Alguien está hablando...</span>`;
                }
            }
            else {
                if (!estaTransmitiendo && !btnPtt.classList.contains('recibiendo')) {
                    restaurarEstadoOriginal();
                }
            }
        });
    }

    function reproducirVozEntrante(base64Data) {
        const audio = new Audio(base64Data);
        audio.volume = 1.0;
        audio.play().then(() => {
            audio.onended = function() { restaurarEstadoOriginal(); };
        }).catch(e => restaurarEstadoOriginal());
    }

    function restaurarEstadoOriginal() {
        btnPtt.classList.remove('recibiendo');
        lblEstado.innerText = "ONLINE - FRECUENCIA ABIERTA";
        lblQuienHabla.innerHTML = `<span style="color: var(--co-texto-secundario);"><i class="fa-solid fa-circle-check"></i> Frecuencia Lista (Silencio)</span>`;
    }

    // --- ACCIÓN FÍSICA MANTENER PRESIONADO ---
    btnPtt.addEventListener('mousedown', iniciarTransmision);
    window.addEventListener('mouseup', detenerTransmision);
    btnPtt.addEventListener('touchstart', (e) => { e.preventDefault(); iniciarTransmision(); });
    window.addEventListener('touchend', detenerTransmision);

    function iniciarTransmision() {
        if (btnPtt.classList.contains('recibiendo') || estaTransmitiendo || !streamMicrofono) return;
        
        estaTransmitiendo = true;
        fragmentosAudio = [];
        btnPtt.classList.add('transmitiendo');
        txtBotonPtt.innerText = "AL AIRE";
        lblEstado.innerText = "TRANSMITIENDO...";
        lblQuienHabla.innerHTML = `<span style="color:var(--co-transmision); font-weight:600;"><i class="fa-solid fa-microphone-lines"></i> Hablando...</span>`;
        
        db.ref('radio_frecuencia/canal_01/estado').set({ transmitiendo: true, usuario: miCodigoSocio });

        try {
            grabadorNativo = new MediaRecorder(streamMicrofono);
            grabadorNativo.ondataavailable = (e) => { if (e.data.size > 0) fragmentosAudio.push(e.data); };
            
            grabadorNativo.onstop = () => {
                if (fragmentosAudio.length > 0) {
                    const blobAudio = new Blob(fragmentosAudio, { type: 'audio/webm' });
                    const lector = new FileReader();
                    lector.readAsDataURL(blobAudio);
                    lector.onloadend = () => {
                        db.ref('radio_frecuencia/canal_01').set({
                            usuario: miCodigoSocio,
                            audioBase64: lector.result,
                            timestamp: firebase.database.ServerValue.TIMESTAMP,
                            estado: { transmitiendo: false, usuario: "" }
                        });
                    };
                } else {
                    db.ref('radio_frecuencia/canal_01/estado').set({ transmitiendo: false, usuario: "" });
                }
            };
            grabadorNativo.start();
        } catch(err) {
            detenerTransmision();
        }
    }

    function detenerTransmision() {
        if (!estaTransmitiendo) return;
        estaTransmitiendo = false;
        btnPtt.classList.remove('transmitiendo');
        txtBotonPtt.innerText = "PULSAR";
        lblEstado.innerText = "ONLINE - FRECUENCIA ABIERTA";

        if (grabadorNativo && grabadorNativo.state !== "inactive") {
            try { grabadorNativo.stop(); } catch(e){}
        } else {
            db.ref('radio_frecuencia/canal_01/estado').set({ transmitiendo: false, usuario: "" });
        }
    }
